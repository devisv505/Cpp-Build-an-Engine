The Application & Window Classes

Now we turn the GLFW code from the platform chapter into reusable engine classes. Window owns the native desktop window on both Windows and macOS, while Application owns the frame loop and gives the game clean update and render hooks.

What you'll learn
  • Use RAII to own a GLFWwindow*.
  • Expose logical and framebuffer sizes correctly on Retina displays.
  • Build a portable Application lifecycle with std::chrono.
  • Keep the native graphics surface inside the renderer backend.

A portable Window interface

#pragma once
#include <cstdint>
#include <string>

struct GLFWwindow;

class Window {
public:
    Window() = default;
    ~Window();

    Window(const Window&) = delete;
    Window& operator=(const Window&) = delete;

    bool Create(const std::string& title, int width, int height);
    void PollEvents() const;
    bool ShouldClose() const;
    void GetFramebufferSize(int& width, int& height) const;
    GLFWwindow* NativeHandle() const { return m_handle; }

private:
    GLFWwindow* m_handle = nullptr;
};

The forward declaration keeps GLFW out of most headers. Only the platform implementation includes <GLFW/glfw3.h>.

Implementing Window with GLFW

#include "Window.h"
#include <GLFW/glfw3.h>
#include <iostream>

Window::~Window() {
    if (m_handle) glfwDestroyWindow(m_handle);
    glfwTerminate();
}

bool Window::Create(const std::string& title, int width, int height) {
    if (!glfwInit()) {
        std::cerr << "GLFW initialization failed\n";
        return false;
    }

    // The renderer creates a Direct3D swap chain or Metal layer later.
    glfwWindowHint(GLFW_CLIENT_API, GLFW_NO_API);
    m_handle = glfwCreateWindow(width, height, title.c_str(), nullptr, nullptr);
    return m_handle != nullptr;
}

void Window::PollEvents() const { glfwPollEvents(); }

bool Window::ShouldClose() const {
    return !m_handle || glfwWindowShouldClose(m_handle);
}

void Window::GetFramebufferSize(int& width, int& height) const {
    glfwGetFramebufferSize(m_handle, &width, &height);
}
Framebuffer size matters on macOS

A 1280×720 window can have a 2560×1440 framebuffer on a Retina display. Give the renderer the framebuffer size, not the logical window size, when creating or resizing render targets.

The Application lifecycle

#pragma once
#include "Window.h"
#include <string>

class Application {
public:
    Application(std::string title, int width, int height);
    virtual ~Application() = default;
    int Run();

protected:
    virtual bool OnInitialize() { return true; }
    virtual void OnUpdate(float deltaTime) {}
    virtual void OnRender() {}
    virtual void OnShutdown() {}

    Window& GetWindow() { return m_window; }

private:
    Window m_window;
    std::string m_title;
    int m_width;
    int m_height;
};
#include "Application.h"
#include <chrono>

Application::Application(std::string title, int width, int height)
    : m_title(std::move(title)), m_width(width), m_height(height) {}

int Application::Run() {
    if (!m_window.Create(m_title, m_width, m_height)) return 1;
    if (!OnInitialize()) return 1;

    using Clock = std::chrono::steady_clock;
    auto previous = Clock::now();

    while (!m_window.ShouldClose()) {
        m_window.PollEvents();

        const auto now = Clock::now();
        const float dt = std::chrono::duration<float>(now - previous).count();
        previous = now;

        OnUpdate(dt);
        OnRender();
    }

    OnShutdown();
    return 0;
}

Using the framework

#include "Application.h"

class Game : public Application {
public:
    Game() : Application("My Engine", 1280, 720) {}

protected:
    bool OnInitialize() override { return true; }
    void OnUpdate(float dt) override { m_time += dt; }
    void OnRender() override { /* renderer comes next */ }

private:
    float m_time = 0.0f;
};

int main() {
    Game game;
    return game.Run();
}

Only the selected renderer backend should include glfw3native.h. Game code continues to see the same Window and Application classes.

Recap
  • Window owns GLFW with RAII and works on Windows and macOS.
  • Application owns the lifecycle and calls portable update/render hooks.
  • std::chrono::steady_clock produces frame-rate-independent delta time.
  • The renderer uses framebuffer pixels and keeps native window APIs in backend files.
